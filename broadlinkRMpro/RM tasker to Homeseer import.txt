Navn = tegn(0)
Kommando = tegn(1)

dvRef = hs.GetDeviceRefByName(Navn)

if dvRef <= 0 then
   dv = hs.NewDeviceEx(Navn)

 dv = hs.GetDeviceByRef(dvRef)

 EvRef = hs.NewEventEx(Kommando, Navn,"Kommando")
